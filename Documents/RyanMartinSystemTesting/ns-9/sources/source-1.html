     private boolean ifActive;
     private int roleId;
 
     @OneToMany(mappedBy = "user")

     private List<Bookmark> bookmarks;

         this.name = name;

         this.ifActive = false;
     }

     }

     }

         this.password = password;

         return emailId;

         this.emailId = emailId;
